impl<'t> RequestBuilder<'t, Domain>

fn delete(self) -> RequestBuilder<'t, HeaderOnly>

Returns a request that can be used to an existing domain. Returns a header with "status: 204 No Content" if successful

Example

// ... domgr set up same as before
match domgr.domain("super.com")
           .delete()
           .retrieve() {
    Ok(_)  => println!("Success"),
    Err(_) => println!("Error")
}
